fixes F :: "('trm ⇒ 'trm) set"
  and encF :: "('trm ⇒ 'trm) ⇒ 'fmla" 
  and N :: "'trm ⇒ 'trm"
  and ssap :: "'fmla ⇒ 'trm ⇒ 'trm"

assumes 
(* For the members f of F, we only care about their action on numerals: *)
F[simp,intro!]: "⋀ f n. f ∈ F ⟹ n ∈ num ⟹ f n ∈ num"
and
encF[simp,intro!]: "⋀ f. f ∈ F ⟹ encF f ∈ ptrm (Suc 0)"
and 
N[simp,intro!]: "N ∈ F" 
and 
ssap[simp]: "⋀φ. φ ∈ fmla ⟹ Fvars φ = {inp} ⟹ ssap φ ∈ F"
(*  *)
and 
ReprF: "⋀f n. f ∈ F ⟹ n ∈ num ⟹ prveqlPT (instInp (encF f) n) (f n)"
and
CapN: "⋀φ. φ ∈ fmla ⟹ Fvars φ = {} ⟹ N ⟨φ⟩ = ⟨neg φ⟩"
and  
CapSS: (* We consider formulas ψ of one variable, called "inp": *)
"⋀ ψ f. ψ ∈ fmla ⟹ Fvars ψ = {inp} ⟹ f ∈ F ⟹ 
    ssap ψ ⟨encF f⟩ = ⟨cmpP ψ 0 (instInp (encF f) ⟨encF f⟩)⟩" 
begin 

lemma encF_fmla[simp,intro!]: "⋀ f. f ∈ F ⟹ encF f ∈ fmla"
by auto

lemma enc_trm: "φ ∈ fmla ⟹ ⟨φ⟩ ∈ trm"
by auto
  

definition τJ :: "'fmla ⇒ 'fmla" where 
"τJ ψ ≡ instInp (encF (ssap ψ)) (⟨encF (ssap ψ)⟩)"

definition φJ :: "'fmla ⇒ 'fmla" where 
"φJ ψ ≡ cmpP ψ 0 (τJ ψ)" 

lemma τJ[simp]: 
assumes "ψ ∈ fmla" and "Fvars ψ = {inp}"
shows "τJ ψ ∈ ptrm 0"
unfolding τJ_def apply(rule instInp)
using assms by auto

lemma τJ_fmla[simp]: 
assumes "ψ ∈ fmla" and "Fvars ψ = {inp}"
shows "τJ ψ ∈ fmla"
using τJ[OF assms] unfolding ptrm_def by auto

lemma FvarsT_τJ[simp]: 
assumes "ψ ∈ fmla" and "Fvars ψ = {inp}"
shows "Fvars (τJ ψ) = {out}"
using τJ[OF assms] unfolding ptrm_def by auto

lemma φJ[simp]: 
assumes "ψ ∈ fmla" and "Fvars ψ = {inp}"
shows "φJ ψ ∈ fmla"
unfolding φJ_def apply(rule cmpP_fmla)
using assms by auto

lemma Fvars_φJ[simp]: 
assumes "ψ ∈ fmla" and "Fvars ψ = {inp}"
shows "Fvars (φJ ψ) = {}"
using assms unfolding φJ_def by (auto simp: Fvars_cmpP2)

lemma diagonalization: 
assumes ψ[simp]: "ψ ∈ fmla" and [simp]: "Fvars ψ = {inp}"
shows "prveqlPT (τJ ψ) ⟨cmpP ψ 0 (τJ ψ)⟩ ∧ 
       prv (eqv (φJ ψ) (instInp ψ ⟨φJ ψ⟩))"
proof 
  define f where "f ≡ ssap ψ" 
  have f[simp]: "f ∈ F" unfolding f_def using assms by auto
  have ff: "f ⟨encF f⟩ = ⟨cmpP ψ 0 (τJ ψ)⟩"
  using assms unfolding f_def τJ_def by (intro CapSS) auto

  show "prveqlPT (τJ ψ) ⟨cmpP ψ 0 (τJ ψ)⟩" 
  using ReprF[OF f, of "⟨encF f⟩"]  
  unfolding τJ_def[of ψ, unfolded f_def[symmetric],symmetric] ff[symmetric]
  by auto 
  from prveqlPT_prv_instInp_eqv_cmpP[OF ψ, of "τJ ψ", OF _ _ _ _ this,
           unfolded φJ_def[symmetric]]
  show "prv (eqv (φJ ψ) (instInp ψ ⟨φJ ψ⟩))"
  by auto  
qed 


end ― ‹context Jeroslow_Diagonalization›

begin

(* A version of HBL1 that uses the inp variable: *)
lemma HBL1_inp: 
"φ ∈ fmla ⟹ Fvars φ = {} ⟹ prv φ ⟹ prv (instInp Pinp ⟨φ⟩)"
unfolding Pinp_def instInp_def by (auto intro: HBL1)
  
(* Consistency formula a la Jeroslow: *)
definition jcons :: "'fmla" where 
"jcons ≡ all xx (neg (cnj (instInp Pinp (Var xx)) 
                           (cmpP Pinp 0 (instInp (encF N) (Var (xx))))))" 

lemma prv_eql_subst_trm3:
"x ∈ var ⟹ φ ∈ fmla ⟹ t1 ∈ trm ⟹ t2 ∈ trm ⟹ 
prv (eql t1 t2) ⟹ prv (subst φ t1 x) ⟹ prv (subst φ t2 x)"
using prv_eql_subst_trm2  
  by (meson subst prv_imp_mp) 

lemma Pinp[simp,intro!]: "Pinp ∈ fmla" 
and Fvars_Pinp[simp]: "Fvars Pinp = {inp}"
unfolding Pinp_def by (auto simp: Fvars_subst)

lemma ReprF_combineWith_CapN: 
assumes "φ ∈ fmla" and "Fvars φ = {}"
shows "prveqlPT (instInp (encF N) ⟨φ⟩) ⟨neg φ⟩" 
using assms unfolding CapN[symmetric, OF assms] by (intro ReprF) auto  
 
theorem jeroslow_godel_second: 
assumes consistent
(* Implicit unreasonable assumption under which Jeroslow's formulation works: *)
assumes unreasonable:
        "let ψ = cmpP Pinp (Suc 0) (encF N); 
             τ = τJ ψ; 
             φ = cmpP (cmpP Pinp (Suc 0) (encF N)) 0 τ;  
             φ' = cmpP Pinp 0 (cmpP (encF N) 0 τ) 
         in prv (eql ⟨φ⟩ ⟨φ'⟩)"
shows "¬ prv jcons"
proof 
  assume *: "prv jcons"

  define ψ where "ψ ≡ cmpP Pinp (Suc 0) (encF N)"
  define τ where "τ ≡ τJ ψ"
  define φ where "φ ≡ φJ ψ"
  have ψ[simp,intro]: "ψ ∈ fmla" "Fvars ψ = {inp}" 
  unfolding ψ_def by (auto simp: Fvars_cmpP2) 
  have τ[simp,intro]: "τ ∈ ptrm 0" "τ ∈ fmla" "Fvars τ = {out}" 
    unfolding τ_def by auto
  have [simp]: "φ ∈ fmla" "Fvars φ = {}" unfolding φ_def by auto

  define eNτ where "eNτ ≡ cmpP (encF N) 0 τ"
  have eNτ[simp]: "eNτ ∈ ptrm 0" "eNτ ∈ fmla" "Fvars eNτ = {out}"
   unfolding eNτ_def by (auto simp: cmpP1 Fvars_cmpP)
  define φ' where "φ' ≡ cmpP Pinp 0 eNτ" 
  have [simp]: "φ' ∈ fmla" "Fvars φ' = {}" unfolding φ'_def by (auto simp: Fvars_cmpP)

  have φφ': "prv (imp φ φ')" and φ'φ: "prv (imp φ' φ)" and φeφ': "prv (eqv φ φ')"
   unfolding φ_def φJ_def φ'_def eNτ_def τ_def[symmetric] unfolding ψ_def
   using prv_cmpP_compose[of Pinp "encF N" τ] by auto

  from diagonalization[OF ψ]
  have "prveqlPT τ ⟨cmpP ψ 0 τ⟩" and **: "prv (eqv φ (instInp ψ ⟨φ⟩))" 
  unfolding τ_def[symmetric] φ_def[symmetric] by auto
  have "**1": "prv (imp φ (instInp ψ ⟨φ⟩))" "prv (imp (instInp ψ ⟨φ⟩) φ)"
   using prv_imp_eqvEL[OF _ _ **] prv_imp_eqvER[OF _ _ **] by auto

  from SHBL3[OF eNτ(1)] 
  have "prv (imp (cmpP Pinp 0 eNτ) (instInp Pinp ⟨cmpP Pinp 0 eNτ⟩))" .
  hence "prv (imp φ' (instInp Pinp ⟨φ'⟩))" unfolding φ'_def .
  from prv_prv_imp_trans[OF _ _ _ φφ' this]
  have 0: "prv (imp φ (instInp Pinp ⟨φ'⟩))" by auto

  note unr = unreasonable[unfolded Let_def 
    φ_def[unfolded φJ_def τ_def[symmetric], symmetric] ψ_def[symmetric] 
      τ_def[symmetric] eNτ_def[symmetric] φ'_def[symmetric] φJ_def] 

  have 1: "prv (imp φ (instInp Pinp ⟨φ⟩))"
  apply(rule nprv_prvI, auto)
  apply(rule nprv_impI, auto)
  apply(rule nprv_addLemmaE[OF unr], auto)
  apply(rule nprv_addImpLemmaE[OF 0], auto intro: nprv_hyp)
  apply(rule nprv_clear3_3, auto)
  apply(rule nprv_eql_substE01[of "⟨φ⟩" "⟨φ'⟩" _ "instInp Pinp (Var xx)" xx], 
     auto simp: instInp_def nprv_hyp) . 

  have 2: "prv (imp φ (cnj (instInp Pinp ⟨φ⟩) 
                           (instInp ψ ⟨φ⟩)))"
  apply(rule nprv_prvI, auto)
  apply(rule nprv_impI, auto)
  apply(rule nprv_cnjI, auto)
  subgoal apply(rule nprv_addImpLemmaE[OF 1], auto intro: nprv_hyp) .
  subgoal apply(rule nprv_addImpLemmaE[OF "**1"(1)], auto intro: nprv_hyp) . .

  define z where "z ≡ Variable (Suc (Suc 0))" 
  have z_facts[simp]: "z ∈ var" "z ≠ xx" "z ∉ Fvars Pinp"
    "out ≠ z ∧ z ≠ out" "inp ≠ z ∧ z ≠ inp" 
   unfolding z_def by auto
  have 30: "subst (cmpP Pinp 0 (instInp (encF N) (Var xx))) ⟨φ⟩ xx = 
            cmpP Pinp 0 (instInp (encF N) ⟨φ⟩)" 
  unfolding z_def[symmetric] instInp_def cmpP_def Let_def apply auto 
  apply(subst subst_compose_diff, auto)
  apply(subst subst_subst, auto simp: Fvars_subst)
  apply(subst subst_notIn[of _ _ xx], auto simp: Fvars_subst)
  apply(subst subst_compose_diff, auto) . 

  have 31: "subst (instInp Pinp (Var xx)) ⟨φ⟩ xx = 
            instInp Pinp ⟨φ⟩" unfolding instInp_def by auto
              

  have 3: "prv (neg (cnj (instInp Pinp ⟨φ⟩) 
                         (instInp ψ ⟨φ⟩)))"
  apply(rule nprv_prvI, auto)
  apply(rule nprv_addLemmaE[OF *, unfolded jcons_def], auto)
  apply(rule nprv_allE0[of _ _ _ "⟨φ⟩"], auto  simp: 30 31)
  apply(rule nprv_clear2_2, auto)
  apply(rule nprv_negI, auto)
  apply(rule nprv_negE0, auto)
  apply(rule nprv_clear2_2, auto)
  apply(rule nprv_cnjE0, auto)
  apply(rule nprv_clear3_3, auto)
  apply(rule nprv_cnjI, auto)
  subgoal 
    apply(auto intro: nprv_hyp) .
  subgoal 
    apply(rule nprv_clear2_1, auto)
    apply(auto simp: nprv_hyp instInp_cmpP ψ_def) . .

  have ***: "prv (neg φ)"
  apply(rule nprv_prvI, auto) 
  apply(rule nprv_negI, auto) 
  apply(rule nprv_addImpLemmaE[OF 2], auto simp: nprv_hyp)
  apply(rule nprv_addLemmaE[OF 3], auto)
  apply(rule nprv_negE0, auto simp: nprv_hyp) .

  have 4: "prv (instInp Pinp ⟨neg φ⟩)" using HBL1_inp[OF _ _ ***] by auto

  have 5: "prveqlPT (instInp (encF N) ⟨φ⟩) ⟨neg φ⟩"
    using ReprF_combineWith_CapN[of φ] by auto

  have 6: "prv (cmpP Pinp 0 (instInp (encF N) ⟨φ⟩))"
  apply(rule nprv_prvI, auto)
  apply(rule nprv_addLemmaE[OF 4], auto)
  apply(rule prveqlPT_nprv_cmpP_instInp[OF _ _ _ _ _ 5], auto simp: instInp) .

  note lem = "**1"(2)[unfolded ψ_def]
  have "prv φ"
  apply(rule nprv_prvI, auto)
  apply(rule nprv_addLemmaE[OF 6], auto)
  apply(rule nprv_addImpLemmaE[OF lem], auto simp: nprv_hyp instInp_cmpP) .

  from this *** `consistent` show False unfolding consistent_def3 by auto
qed
